- Validating floating point negative number
        user_input = input("Enter a number: ")

        # Remove one occurrence of the decimal point
        temp_input = user_input.replace(".", "", 1)

        # If the number is negative, remove the leading "-"
        if temp_input.startswith("-"):
           temp_input = temp_input[1:]

        # Check if the remaining characters are digits
        if temp_input.isdigit():
           number = float(user_input)
           print(f"Valid float: {number}")
        else:
           print("Invalid input! Please enter a valid number.")
